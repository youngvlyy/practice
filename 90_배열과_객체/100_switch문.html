switch() 문은 다중 if문과 매우 비슷하다.

        범위에 대한 조건식은 사용할 수 없다.
        비교할 변수의 값을 이용하여 분기처리 함.

        switch(조건비교할 변수){
            case 비교값1 : 
                실행할 코드;
                break;
            
            case 비교값1 : 
                실행할 코드;
                break;
            
            case 비교값1 : 
                실행할 코드;
                break;
           
            default:
                실행할 코드;
        }
